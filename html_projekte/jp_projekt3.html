📊 JS :: Projekt 3 :: Notenschnitt berechnen

🎬 1 Teaser

In diesem Projekt kannst du deinen Notendurchschnitt berechnen. Gib einfach deine Noten ein und drücke auf "Berechnen".

Das Tool zeigt dir den Durchschnitt deiner Noten an. Du kannst beliebig viele Noten hinzufügen. Viel Erfolg!

Füge jeweils pro Modul den JavaScript-Code (jp_notenschnitt.js) und den HTML-Code (jp_notenschnitt.html) ein. Schritt für Schritt näherst du dich dann dem Programm.

Verwende den CSS-Code, der am Ende der Seite angegeben wird, und lege diesen in der Datei `jp_projekte.css` ab.

Hier ist der komplette JavaScript-Code:

            
                let sum = 0;
                let count = 0;

                function addGrade(grade) {
                    sum += grade;
                    count++;
                }

                function calculateAverage(sum, count) {
                    return (sum / count).toFixed(2);
                }

                function getInputAndCalculate() {
                    const inputValue = parseInt($('#firstNumber').val(), 10);

                    if (isNaN(inputValue) || inputValue < 1 || inputValue > 6) {
                        updateResultDisplay('Bitte nur Zahlen zwischen 1 und 6 eintragen.', 'error');
                        return;
                    }

                    addGrade(inputValue);
                    const average = calculateAverage(sum, count);
                    updateResultDisplay(`Durchschnitt: ${average}. Anzahl Noten: ${count}`, 'success');
                }

                function updateResultDisplay(message, type) {
                    const resultElement = $('#showCallbackOfChallenge');
                    resultElement.css('visibility', 'visible');

                    if (type === 'error') {
                        resultElement.css('background-color', '#c22b67');
                    } else if (type === 'success') {
                        resultElement.css('background-color', '#28a745');
                    }

                    resultElement.html(message);
                }

                $(document).ready(function () {
                    $('#sozStartButton').click(getInputAndCalculate);
                });
